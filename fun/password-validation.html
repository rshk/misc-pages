$('#form').submit(function(e){
  e.preventDefault();
  var passw = $('#input-password').val();
  var answers = {};
  for (var i=1; i<=3; i++) {
    answers[i] = $('#input-question-' + i).val();
  }
  if (passw == answers[1]) {
	alert("Only stupids use their birthday as password!");
  }
  else if (passw == answers[2]) {
	alert("You can't use your cat's name as password!");
  }
  else if (passw == answers[3]) {
	alert("Seriously? Your mom's name as password?");
  }
});
